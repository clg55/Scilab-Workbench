Description

Given vector valued inputs this block merges inputs in an single output vector. So , where are numbered from top to bottom. Input and Output port sizes are determined by the context.

Computational function (type 4) 


#include "scicos_block.h"
#include <math.h>

void multiplex(scicos_block *block,int flag)
{
  int i,j,k;
  if (block->nin==1){
    k=0;
    for (i=0;i<block->nout;++i){
      for (j=0;j<block->outsz[i];++j){
	block->outptr[i][j]=block->inptr[0][k];
	++k;
      }
    }
  }else {
    k=0;
    for (i=0;i<block->nin;++i){
      for (j=0;j<block->insz[i];++j){
	block->outptr[0][k]=block->inptr[i][j];
	++k;
      }
    }
  }
}
